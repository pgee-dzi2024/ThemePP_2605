Проект: „Анализатор на отзиви“ (Sentiment Analysis уеб приложение)
Приложението приема текст от потребителя (и по желание – извлича текст от URL), прави предобработка (NLP), после класифицира настроението като „Позитивен / Негативен / Неутрален“ и показва оценка/вероятност + визуализация (напр. лента/прогрес бар и графика).
Подходяща архитектура за дипломна работа:
•	Backend: Django + DRF (API)
•	Frontend:
o	вариант A (най-лесно): Django templates + малко JS (fetch към DRF)
o	вариант B: отделен фронтенд (React/Vue) – само ако имаш време
•	NLP модел:
o	най-лесно за демонстрация: готов модел (напр. transformers pipeline) или TextBlob/VADER (ако е английски)
o	ако искаш по-„училищно/ML“: sklearn (TF IDF + LogisticRegression) обучен на dataset
Тъй като в заданието пише „потребителски текст и/или извличане на текст от URL“ и „визуализация“, най-добре е да направиш:
•	вход: textarea + поле за URL (по избор)
•	изход: label (клас) + confidence score + графика (bar chart) на вероятностите за 3-те класа
•	история на анализите (запазване в база) – много добре стои като функционалност и за демонстрация

import re

def clean_text(text: str) -> str:
    """
    Лека нормализация (достатъчна за дипломната):
    - trim
    - премахване на множество whitespace
    - махане на "празни" символи
    """
    if not text:
        return ""

    t = text.strip()

    # normalize whitespace
    t = re.sub(r"\s+", " ", t)

    # remove zero-width chars sometimes present in copy/paste
    t = t.replace("\u200b", "").replace("\ufeff", "")

    return t

Стъпка 3.3: main/nlp/sentiment.py
Тук ползваме transformers.pipeline. Ще направим:
•	lazy load (зарежда модела веднъж при първо извикване)
•	функция predict_sentiment(text) → връща label + scores за 3 класа
•	„NEUTRAL“ чрез праг: ако моделът не е достатъчно уверен, маркираме неутрално
from transformers import pipeline
from .preprocess import clean_text

# Lazy-loaded singleton
_classifier = None

def get_classifier():
    global _classifier
    if _classifier is None:
        # Multilingual sentiment model (примерен избор).
        # Ако искаш, после можем да го сменим с друг multilingual модел.
        model_name = "cardiffnlp/twitter-xlm-roberta-base-sentiment"
        _classifier = pipeline(
            "sentiment-analysis",
            model=model_name,
            tokenizer=model_name,
            return_all_scores=True
        )
    return _classifier


def predict_sentiment(text: str, neutral_threshold: float = 0.60) -> dict:
    """
    Връща:
      {
        "sentiment": "positive|negative|neutral",
        "scores": {"positive": x, "neutral": y, "negative": z},
        "clean_text": "..."
      }

    neutral_threshold:
      ако най-високата вероятност е под този праг -> neutral
    """
    cleaned = clean_text(text)
    if not cleaned:
        return {
            "sentiment": "neutral",
            "scores": {"positive": 0.0, "neutral": 1.0, "negative": 0.0},
            "clean_text": cleaned,
        }

    clf = get_classifier()

    # pipeline с return_all_scores=True връща списък с резултати, напр:
    # [[{'label': 'positive', 'score': 0.8}, {'label':'neutral',...}, ...]]
    raw = clf(cleaned)[0]

    # Нормализиране на етикетите към positive/neutral/negative
    scores = {"positive": 0.0, "neutral": 0.0, "negative": 0.0}
    for item in raw:
        lbl = item["label"].lower()

        # Модели връщат различни имена: LABEL_0/1/2 или positive/neutral/negative
        # За този модел обикновено са: negative/neutral/positive
        if "pos" in lbl or lbl == "positive":
            scores["positive"] = float(item["score"])
        elif "neg" in lbl or lbl == "negative":
            scores["negative"] = float(item["score"])
        elif "neu" in lbl or lbl == "neutral":
            scores["neutral"] = float(item["score"])

    # Ако моделът не върне neutral (в някои случаи), ще остане 0.0 — това не е проблем,
    # защото пак ще изчислим крайния label логично.
    best_label = max(scores, key=scores.get)
    best_score = scores[best_label]

    if best_score < neutral_threshold:
        sentiment = "neutral"
        # Ако сме го принудили към neutral, е хубаво да дадем някаква "разумна" неутрална стойност.
        # Нека преразпределим: неутралното = 1 - best_score, останалото както е.
        # (Това е прагова евристика, описва се в дипломната.)
        forced_neu = 1.0 - best_score
        # нормализация (за да сумират ~1)
        total = scores["positive"] + scores["negative"] + forced_neu
        if total > 0:
            scores = {
                "positive": scores["positive"] / total,
                "negative": scores["negative"] / total,
                "neutral": forced_neu / total,
            }
    else:
        sentiment = best_label

    return {"sentiment": sentiment, "scores": scores, "clean_text": cleaned}
Забележка: конкретният модел cardiffnlp/twitter-xlm-roberta-base-sentiment обикновено връща 3 класа (neg/neu/pos). Евристиката за neutral_threshold ти дава контрол и изглежда добре в дипломната.

from rest_framework.response import Response
from rest_framework import status

from .serializers import AnalyzeTextSerializer
from .models import ReviewAnalysis
from .nlp.sentiment import predict_sentiment


class AnalyzeTextAPIView(APIView):
    """
    POST /api/analyze/
    Body: {"text": "..."}
    Response: sentiment + scores + id
    """
    def post(self, request):
        serializer = AnalyzeTextSerializer(data=request.data)
        serializer.is_valid(raise_exception=True)

        text = serializer.validated_data["text"]
        result = predict_sentiment(text)

        scores = result["scores"]
        sentiment = result["sentiment"]
        clean_text = result["clean_text"]

        obj = ReviewAnalysis.objects.create(
            input_text=text,
            clean_text=clean_text,
            sentiment=sentiment,
            pos_score=scores.get("positive", 0.0),
            neu_score=scores.get("neutral", 0.0),
            neg_score=scores.get("negative", 0.0),
        )

        return Response(
            {
                "id": obj.id,
                "sentiment": sentiment,
                "scores": {
                    "positive": obj.pos_score,
                    "neutral": obj.neu_score,
                    "negative": obj.neg_score,
                },
                "clean_text": clean_text,
            },
            status=status.HTTP_200_OK
        )

from transformers import pipeline
from .preprocess import clean_text

_classifier = None

def get_classifier():
    global _classifier
    if _classifier is None:
        model_name = "cardiffnlp/twitter-xlm-roberta-base-sentiment"
        _classifier = pipeline(
            "sentiment-analysis",
            model=model_name,
            tokenizer=model_name,
            return_all_scores=True
        )
    return _classifier


def _map_label_to_sentiment(label: str) -> str:
    """
    Мапване за този модел:
    LABEL_0 -> negative
    LABEL_1 -> neutral
    LABEL_2 -> positive
    """
    l = label.upper().strip()
    if l == "LABEL_0":
        return "negative"
    if l == "LABEL_1":
        return "neutral"
    if l == "LABEL_2":
        return "positive"

    # fallback, ако модел върне текстови етикети
    low = label.lower()
    if "neg" in low:
        return "negative"
    if "neu" in low:
        return "neutral"
    if "pos" in low:
        return "positive"

    return "neutral"


def predict_sentiment(text: str, neutral_threshold: float = 0.60) -> dict:
    cleaned = clean_text(text)
    if not cleaned:
        return {
            "sentiment": "neutral",
            "scores": {"positive": 0.0, "neutral": 1.0, "negative": 0.0},
            "clean_text": cleaned,
        }

    clf = get_classifier()
    raw = clf(cleaned)[0]

    scores = {"positive": 0.0, "neutral": 0.0, "negative": 0.0}
    for item in raw:
        s = _map_label_to_sentiment(item["label"])
        scores[s] = float(item["score"])

    best_label = max(scores, key=scores.get)
    best_score = scores[best_label]

    # евристика за "неутрално", ако моделът не е уверен
    if best_score < neutral_threshold:
        sentiment = "neutral"
    else:
        sentiment = best_label

    return {"sentiment": sentiment, "scores": scores, "clean_text": cleaned}

    # 1) list[dict] (ако top_k е число)
    # 2) list[list[dict]] (ако top_k=None -> всички класове)
    if isinstance(out, list) and len(out) > 0 and isinstance(out[0], list):
        raw = out[0]
    elif isinstance(out, list):
        raw = out
    else:
        raw = []

    scores = {"positive": 0.0, "neutral": 0.0, "negative": 0.0}
    for item in raw:
        # item трябва да е dict: {'label': 'LABEL_2', 'score': 0.9}
        if not isinstance(item, dict):
            continue
        s = _map_label_to_sentiment(item.get("label", ""))
        scores[s] = float(item.get("score", 0.0))

    best_label = max(scores, key=scores.get)
    best_score = scores[best_label]

    sentiment = "neutral" if best_score < neutral_threshold else best_label

    return {"sentiment": sentiment, "scores": scores, "clean_text": cleaned}

    let sentimentChart = null;

    document.getElementById('analyze-form').addEventListener('submit', async function(e) {
        e.preventDefault();

        const textInput = document.getElementById('text-input').value;
        const submitBtn = document.getElementById('submit-btn');
        const csrfToken = document.querySelector('[name=csrfmiddlewaretoken]').value;
